% \cleardoublepage
% \chapter{Bounding Boxes \label{Bbox}}
% 
% 
% Bounding boxes are used to approximate objects and the only operations 
% you can perform on them is to get their bounds abd to check if two
% bounding boxes overlap. 
% 
% Here comes a typical situation where to use them: In order to check
% if two complicated polygons intersect you first check if their 
% respective bounding boxes intersect. If the answer is negative,
% you are done, otherwise you really have to check if the polygons 
% intersect.
%  
\begin{class} {CGAL_Bbox_3}
\CCsection{3D Bbox}

\definition  An object \CCstyle{b} of the class \classname\ is a bounding 
box in the three-dimensional Euclidean space $\E_3$. This class is not 
templated.


\creation
\creationvariable{b}

\CCstyle{#include <CGAL/Bbox_3.h>}

\constructor{CGAL_Bbox_3(double x_min, double y_min, double z_min,
                         double x_max, double y_max, double z_max);}
            {introduces a bounding box \var\ with lower left corner at 
             \CCstyle{(xmin, ymin, zmin)} and with upper right corner at 
	     \CCstyle{(xmax, ymax, zmax)}.}


\operations
\threecolumns{5cm}{4cm}

\method{bool operator==(const CGAL_Bbox_3 &c) const;}
       {Test for equality: two bounding boxes are equal, if the lower left
        and the upper right corners are equal.}

\hidden \method{bool operator!=(const CGAL_Bbox_3 &q) const;}
       {Test for inequality.}

\method{double                 xmin() const;}
       {}
\method{double                 ymin() const;}
       {}
\method{double                 zmin() const;}
       {}
\method{double                 xmax() const;}
       {}
\method{double                 ymax() const;}
       {}
\method{double                 zmax() const;}
       {}


\method{CGAL_Bbox_3              operator+(const CGAL_Bbox_3 &c) const;}
       {returns the bounding box of \var\ and \CCstyle{c}.}

\function{bool CGAL_do_overlap(const CGAL_Bbox_3 &bb1, const CGAL_Bbox_3 &bb2);}
       {}
\end{class}
